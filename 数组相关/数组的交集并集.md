## 求数组的交集的函数





```
Array.prototype.intersect = function(b) {
  var flip = {};
  var res = [];
  for(var i=0; i< b.length; i++) flip[b[i]] = i;
  for(i=0; i<this.length; i++)
    if(flip[this[i]] != undefined) res.push(this[i]);
  return res;
}
 
a = [1,2,3,4, 'a', 'c'];
b = [2,4,'c'];
document.write(a.intersect(b)); //2,4,c
```
